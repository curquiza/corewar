ASM
===

-- BNF:

VALUE :=		INT
REG_NUMBER :=	INT
REGISTRE :=		r REG_NUMBER // vérifier si le numéro de registre existe,
entre 0 et 16.
DIRECT :=		DIRECT_CHAR (VALUE | LABEL)
INDIRECT :=		VALUE | LABEL

PARAM :=		REGISTRE | DIRECT | INDIRECT

PARAMS :=		PARAM (SEPARATOR_CHAR PARAMS)*
LABEL :=		LABEL_CHAR (LABEL_CHARS)*
OP_CODE :=		INT

COMMENT :=		COMMENT_CHAR STRING
NAME :=			NAME_CMD_STRING STRING
DESC :=			COMMENT_CMD_STRING STRING
INSTR :=		(LABEL)* OP_CODE (PARAMS)* (COMMENT)*

LINE :=			COMMENT | NAME | DESC | INSTR


-- ENCODAGE:
header_t puis
OPCODE OCP PARAMS pour chaque INSTR
